# Set up a local TS playground 
cd /home/lifji/Project/sfap/sfap-ts                                                                                                                                                                                                                                                                                                                                                                                                                                      
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  # 1. init a package + install TypeScript and tsx (a zero-config TS runner)                                                                                                                                                                                                                                                                                                                                                                                               
  npm init -y                                                                                                                                                                                                                                                                                                                                                                                                                                                              
  npm install -D typescript tsx @types/node                                                                                                                                                                                                                                                                                                                                                                                                                                
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  # 2. generate a tsconfig.json                                                                                                                                                                                                                                                                                                                                                                                                                                            
  npx tsc --init                                                                                                                                                                                                                                                                                                                                                                                                                                                           
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  # 3. write your first script                                                                                                                                                                                                                                                                                                                                                                                                                                             
  echo 'const greet = (who: string): string => `Hello, ${who}!`;                                                                                                                                                                                                                                                                                                                                                                                                           
  console.log(greet("TypeScript"));' > hello.ts                                                                                                                                                                                                                                                                                                                                                                                                                            
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  # 4. run it (tsx = fast, no build step)                                                                                                                                                                                                                                                                                                                                                                                                                                  
  npx tsx hello.ts                                                                                                                                                                                                                                                                                                                                                                                                                                                         
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  # 5. type-CHECK without running (catches type errors)                                                                                                                                                                                                                                                                                                                                                                                                                    
  npx tsc --noEmit                                                                                                                                                                                                                                                                                                                                                                                                                                                         
                                                                                                                                                                                                                                                                                                                                                                                                                                                                           
  - npx tsx hello.ts — runs the script instantly, no compile step.                                                                                                                                                                                                                                                                                                                                                                                                         
  - npx tsc --noEmit — type-checks the whole folder and reports errors without producing JS. This is the part that makes it a real learning playground.
